\def\fibonacci#1#2#3{%
\newcount\one \one=#1
\newcount\two \two=#2
\number\one,~\number\two,
\newcount\f
\newcount\n \n=#3 \advance\n-2
\loop
  \f=\two \advance\two\one \one=\f
  \number\two,
  \advance \n -1
\ifnum \n>0 \repeat
\dots}
