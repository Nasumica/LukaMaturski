\def\fibonacci#1#2#3{%
\newcount\o \o=#1
\newcount\t \t=#2
\number\o,~\number\t
\newcount\f
\newcount\n \n=#3 \advance\n-2
\loop
  \f=\t \advance\t\o \o=\f
  , \number\t
  \advance \n -1
\ifnum \n>0 \repeat}
